    <script>
     function convert_to_miles() {
      var km = document.getElementById("Input_KM").value;

      if (km === "" || isNaN(km) || km < 0) {
        alert("please no negative number please postive");
        return;
      }

      var miles = km * 0.621371;
      document.getElementById("result").textContent = 
        km + " kilometers = " + miles.toFixed(2) + " miles";
      
      confirm("Do you want to continue?");
    }//confirem

    function clear_fields() {
      document.getElementById("Input_KM").value = "";
      document.getElementById("result").textContent = "";
    }

     const inputKm = prompt("Enter the number of kilometers:");

    }
